const timeStr = () => {
	const now = new Date();
	const timeArr = [
		now.getFullYear(),
		now.getMonth() + 1,
		now.getDate(),
		now.getHours(),
		now.getMinutes(),
		now.getSeconds()
		// now.getMilliseconds()
	];
	// yyyyMMddHHmmssfff
	const timeString = timeArr.reduce((accumulator, currentValue) => accumulator * 100 + currentValue);
	return `${timeString * 1000 + now.getMilliseconds()}`;
};
